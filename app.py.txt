import firebase_admin
from firebase_admin import credentials
import json
import os

# اقرأ الـ JSON من متغير البيئة
firebase_json = os.environ.get("FIREBASE_CREDENTIALS")

# حول النص إلى dict
cred_dict = json.loads(firebase_json)

# فعّل Firebase
cred = credentials.Certificate(cred_dict)
firebase_admin.initialize_app(cred)

db = firestore.client()

st.set_page_config(page_title="مختبر الدواجن", page_icon="🐔")
st.title("🐔 نظام إدارة مزارع الدواجن")

# فورم لإدخال بيانات المزرعة
with st.form("farm_form"):
    name = st.text_input("اسم المزرعة")
    weight = st.number_input("الوزن الحالي (كغم)", min_value=0.0)
    health_status = st.selectbox("الحالة الصحية", ["جيدة", "متوسطة", "حرجة"])
    lab_results = st.text_area("نتائج التحاليل")
    report = st.text_area("التقرير الصحي")

    submit = st.form_submit_button("إرسال ✅")
    if submit:
        db.collection("poultry_farms").add({
            "name": name,
            "weight": weight,
            "health_status": health_status,
            "lab_results": lab_results,
            "report": report
        })
        st.success("✅ تم حفظ البيانات بنجاح!")

# عرض بيانات المزارع
st.subheader("📋 قائمة المزارع")
docs = db.collection("poultry_farms").stream()
for doc in docs:
    d = doc.to_dict()
    st.write(f"🏷️ الاسم: {d['name']}")
    st.write(f"⚖️ الوزن: {d['weight']} كغم")
    st.write(f"💊 الحالة: {d['health_status']}")
    st.write(f"🧪 التحاليل: {d['lab_results']}")
    st.write(f"📝 التقرير: {d['report']}")
    st.markdown("---")
